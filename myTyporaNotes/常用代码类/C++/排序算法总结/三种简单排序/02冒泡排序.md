# 算法思想

- 一趟排序，两两比较，如果左边的数大于右边的数就交换

```c++
#include <iostream>
using namespace std;

int main() {
	int arr[] = { 3,5,1,25,6,3,9,0,-1 };
	int len = sizeof(arr) / sizeof(int);

	for (int i = 1; i < len; i++)
	{
		for (int j = 0; j < len - i; j++) {
			if (arr[j] > arr[j + 1])
			{
				int temp = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = temp;
			}
		}
	}

	for (int k = 0; k < len; k++) {
		cout << arr[k] << " ";
	}
	return 0;
}
```
